Users are scoped to a tenant. Each user belongs to exactly one tenant and carries a set of roles that determine their permissions within that tenant's data. All endpoints in this guide require an authenticated session — read operations need `users:read` and mutating operations need `users:write`.

## Register a User

Create a new user account. The `email` must be unique within the tenant.

**`POST /api/v1/users`**

| Field | Type | Required | Description |
|---|---|---|---|
| `email` | string | yes | User's email address (must be unique within the tenant) |
| `password` | string | yes | Plain-text password (min 8 characters); stored as a bcrypt hash |
| `name` | string | yes | Full display name |
| `phone` | string | no | E.164 formatted phone number (e.g., `+14155552671`) |

**Response**

```json
{
  "id": "usr_01HQ...",
  "email": "jane@acme.com",
  "name": "Jane Doe",
  "phone": "+14155552671",
  "status": "active",
  "locale": "en",
  "timezone": "UTC",
  "created_at": "2024-03-01T12:00:00Z",
  "updated_at": "2024-03-01T12:00:00Z"
}
```

Note: the `password` field is never returned in any response.

          return nil, err
      }
      httpReq.Header.Set("Authorization", "Bearer "+accessToken)
      httpReq.Header.Set("Content-Type", "application/json")

          return nil, err
      }
      defer resp.Body.Close()

      var user User
      if err := json.NewDecoder(resp.Body).Decode(&user); err != nil {
          return nil, err
      }
      return &user, nil

          Name:     "Jane Doe",
          Phone:    "+14155552671",
      })
      if err != nil {
          panic(err)
      }
      fmt.Printf("Created user: %s (id: %s)\n", user.Name, user.ID)
  }

    return user;
  }

## List Users

Retrieve a paginated list of users within the tenant. Results are ordered by `created_at` descending by default.

**`GET /api/v1/users`**

| Query parameter | Type | Description |
|---|---|---|
| `status` | string | Filter by status: `active`, `inactive`, or `suspended` |
| `limit` | integer | Maximum results to return (default: `20`, max: `100`) |
| `offset` | integer | Number of records to skip (default: `0`) |

**Response**

```json
{
  "items": [
    {
      "id": "usr_01HQ...",
      "email": "jane@acme.com",
      "name": "Jane Doe",
      "status": "active",
      "created_at": "2024-03-01T12:00:00Z"
    }
  ],
  "total": 15,
  "limit": 20,
  "offset": 0
}
```

<Tabs>

  ```bash

    -H "Authorization: Bearer $ACCESS_TOKEN"
  ```
  </TabItem>

      resp, err := http.DefaultClient.Do(req)
      if err != nil {
          return nil, err

      defer resp.Body.Close()

      var result UserListResponse
      if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
          return nil, err
      }
      return &result, nil
  }

## Get a User

Retrieve a single user's full profile by their ID.

**`GET /api/v1/users/{id}`**

<Tabs>

    -H "Authorization: Bearer $ACCESS_TOKEN"

  </TabItem>
  <TabItem label="Go">
  ```go
  package main

  import (
      "encoding/json"
      "fmt"
      "net/http"
  )

  func getUser(accessToken, userID string) (*User, error) {
      req, err := http.NewRequest(http.MethodGet,
          "http://localhost:8080/api/v1/users/"+userID, nil)
      if err != nil {
          return nil, err
      }
      req.Header.Set("Authorization", "Bearer "+accessToken)

      resp, err := http.DefaultClient.Do(req)
      if err != nil {
          return nil, err
      }
      defer resp.Body.Close()

      if resp.StatusCode == http.StatusNotFound {
          return nil, fmt.Errorf("user %s not found", userID)
      }

      var user User
      if err := json.NewDecoder(resp.Body).Decode(&user); err != nil {
          return nil, err
      }
      return &user, nil
  }
  ```

## Update User Profile

Partially update a user's profile fields. Only the fields provided in the request body are changed.

**`PATCH /api/v1/users/{id}`**

| Field | Type | Description |
|---|---|---|
| `name` | string | Updated full name |
| `phone` | string | E.164 formatted phone number |
| `locale` | string | BCP 47 language tag (e.g., `en`, `es`, `pt-BR`) |
| `timezone` | string | IANA timezone identifier (e.g., `America/New_York`) |

<Tabs>
  <TabItem label="curl">
  ```bash
  curl -X PATCH http://localhost:8080/api/v1/users/usr_01HQ... \
    -H "Authorization: Bearer $ACCESS_TOKEN" \
    -H "Content-Type: application/json" \
    -d '{
      "name": "Jane Smith",
      "locale": "pt-BR",
      "timezone": "America/Sao_Paulo"
    }'
  ```
  </TabItem>
  <TabItem label="Go">

      "net/http"

      Timezone *string `json:"timezone,omitempty"`
  }

      body, _ := json.Marshal(update)

## Change Password

Allow a user to change their own password. The `current_password` is required to prevent unauthorized changes even if an access token is stolen. This endpoint does not require elevated permissions beyond owning the account.

**`POST /api/v1/users/{id}/change-password`**

| Field | Type | Required | Description |
|---|---|---|---|
| `current_password` | string | yes | The user's current plain-text password |
| `new_password` | string | yes | The desired new password (min 8 characters) |

Returns `204 No Content` on success. All existing sessions are invalidated after a password change.

<Tabs>
  <TabItem label="curl">
  ```bash
  curl -X POST http://localhost:8080/api/v1/users/usr_01HQ.../change-password \
    -H "Authorization: Bearer $ACCESS_TOKEN" \
    -H "Content-Type: application/json" \
    -d '{
      "current_password": "s3cur3P@ss!",
      "new_password": "n3wS3cur3P@ss!"
    }'
  ```
  </TabItem>
  <TabItem label="Go">
  ```go
  package main

  import (
      "bytes"
      "encoding/json"
      "fmt"
      "net/http"
  )

  type ChangePasswordRequest struct {
      CurrentPassword string `json:"current_password"`
      NewPassword     string `json:"new_password"`
  }

  func changePassword(accessToken, userID, currentPassword, newPassword string) error {
      body, _ := json.Marshal(ChangePasswordRequest{
          CurrentPassword: currentPassword,
          NewPassword:     newPassword,
      })

      req, err := http.NewRequest(http.MethodPost,
          fmt.Sprintf("http://localhost:8080/api/v1/users/%s/change-password", userID),
          bytes.NewReader(body))
      if err != nil {
          return err
      }
      req.Header.Set("Authorization", "Bearer "+accessToken)
      req.Header.Set("Content-Type", "application/json")

      resp, err := http.DefaultClient.Do(req)
      if err != nil {
          return err
      }
      defer resp.Body.Close()

      if resp.StatusCode == http.StatusUnauthorized {
          return fmt.Errorf("current password is incorrect")
      }
      // 204 No Content on success
      return nil
  }
  ```
  </TabItem>
  <TabItem label="JavaScript">
  ```javascript
  async function changePassword(accessToken, userId, currentPassword, newPassword) {
    const response = await fetch(
      `http://localhost:8080/api/v1/users/${userId}/change-password`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${accessToken}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          current_password: currentPassword,
          new_password: newPassword,
        }),
      }
    );

    if (response.status === 401) {
      throw new Error('Current password is incorrect');
    }
    if (response.status === 404) {
      throw new Error('User not found');
    }
    if (response.status !== 204) {
      throw new Error('Failed to change password');
    }
    // Success — user must log in again with new password
  }
  ```
  </TabItem>
</Tabs>

## Common Errors

| Status | Code | Description |
|---|---|---|
| `409` | `EMAIL_EXISTS` | A user with the given email already exists in this tenant. |
| `404` | `USER_NOT_FOUND` | No user exists with the given ID within the current tenant. |
| `401` | `WRONG_PASSWORD` | The `current_password` provided to `change-password` is incorrect. |
| `422` | `VALIDATION_ERROR` | A field failed validation, such as a password shorter than 8 characters or an invalid phone format. |
| `403` | `FORBIDDEN` | The authenticated user does not have permission to modify this user's account. |

**Error response shape:**

```json
{
  "code": "EMAIL_EXISTS",
  "message": "A user with email 'jane@acme.com' already exists",
  "request_id": "req_01HQ..."
}
```
