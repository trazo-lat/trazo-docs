Trazo modules have three levels of test coverage, each with a distinct purpose and isolation boundary:

| Level | Scope | Database | File Naming |
|---|---|---|---|
| Unit | Single function or method | Mocked | `service_test.go`, `handler_test.go` |
| Integration | Repository layer | Real (test schema) | `repository_integration_test.go` |
| End-to-End | Full HTTP request cycle | Real (test schema) | `handler_e2e_test.go` |

---

## Unit Tests

Unit tests cover service logic in isolation. The `*core.Core` is replaced with a `MockCoreBundle` that provides pre-configured mock implementations for every interface.

### MockCoreBundle

`MockCoreBundle` is a test helper that assembles a `*core.Core` from mock implementations. It lives in `internal/core/testutil/mock_core.go`.

```go
// internal/core/testutil/mock_core.go
package testutil

import (
    "github.com/trazo-dev/trazo-core-service/internal/core"
    "github.com/trazo-dev/trazo-core-service/internal/core/testutil/mocks"
)

// MockCoreBundle holds mock implementations for every Core interface.
type MockCoreBundle struct {
    Permission *mocks.MockPermissionChecker
    Audit      *mocks.MockAuditRecorder
    Events     *mocks.MockEventBus
    Hooks      *mocks.MockHookExecutor
    DB         *mocks.MockDB
    Session    *mocks.MockSession
    Log        *mocks.MockLogger
    Errors     *mocks.MockErrorFactory
    Modules    *mocks.MockModuleRegistry
    Config     *mocks.MockConfigReader
    Storage    *mocks.MockStorage
    Core       *core.Core
}

// NewMockCore constructs a MockCoreBundle with all mocks initialized.
func NewMockCore(t testing.TB) *MockCoreBundle {
    b := &MockCoreBundle{
        Permission: mocks.NewMockPermissionChecker(t),
        Audit:      mocks.NewMockAuditRecorder(t),
        Events:     mocks.NewMockEventBus(t),
        Hooks:      mocks.NewMockHookExecutor(t),
        DB:         mocks.NewMockDB(t),
        Session:    mocks.NewMockSession(t),
        Log:        mocks.NewMockLogger(t),
        Errors:     core.NewRealErrorFactory(), // Use real errors — they have no side effects
        Modules:    mocks.NewMockModuleRegistry(t),
        Config:     mocks.NewMockConfigReader(t),
        Storage:    mocks.NewMockStorage(t),
    }
    b.Core = core.New(
        b.Permission, b.Audit, b.Events, b.Hooks,
        b.DB, b.Session, b.Log, b.Errors,
        b.Modules, b.Config, b.Storage,
    )
    return b
}
```

<Aside type="tip">
  The mock implementations are generated by [mockery](https://github.com/vektra/mockery). Run `go generate ./internal/core/...` to regenerate them after interface changes.
</Aside>

### Service Unit Test

```go
// internal/modules/domain/repairs/service_test.go
package repairs_test

import (
    "context"
    "testing"

    "github.com/google/uuid"
    "github.com/stretchr/testify/assert"
    "github.com/stretchr/testify/mock"
    "github.com/trazo-dev/trazo-core-service/internal/core/testutil"
    "github.com/trazo-dev/trazo-core-service/internal/modules/domain/repairs"
)

func TestRepairsService_Create_Success(t *testing.T) {
    bundle := testutil.NewMockCore(t)
    repo := &mockRepository{} // local mock or testify mock

    // Set up permission check to pass
    bundle.Permission.On("MustCan", mock.Anything, "repairs.create").Return()

    // Set up hooks to be a no-op
    bundle.Hooks.On("Execute", mock.Anything, "repairs.before_create", mock.Anything).
        Return(nil, nil)

    // Set up the transaction to call the closure and succeed
    expectedRepair := &repairs.Repair{
        ID:          uuid.New(),
        VehicleID:   uuid.New(),
        Description: "Brake replacement",
        Status:      "pending",
    }
    bundle.DB.On("Tx", mock.Anything, mock.AnythingOfType("func(core.Tx) error")).
        Run(func(args mock.Arguments) {
            // Execute the closure to simulate a real transaction
            fn := args.Get(1).(func(core.Tx) error)
            mockTx := testutil.NewMockTx(t)
            repo.repair = expectedRepair
            _ = fn(mockTx)
        }).
        Return(nil)

    // Set up event emission inside the transaction
    bundle.Events.On("EmitTx", mock.Anything, mock.Anything, mock.Anything).Return(nil)

    // Audit recording should fire after transaction
    bundle.Audit.On("Create", mock.Anything, mock.Anything, mock.Anything).Return()

    // Logger should produce a log line
    bundle.Log.On("Logger", mock.Anything).Return(testutil.NewNopLogger())

    svc := repairs.NewService(bundle.Core, repo)
    repair, err := svc.Create(context.Background(), repairs.CreateRepairInput{
        VehicleID:   expectedRepair.VehicleID,
        Description: "Brake replacement",
    })

    assert.NoError(t, err)
    assert.NotNil(t, repair)
    assert.Equal(t, "pending", repair.Status)
    bundle.Permission.AssertExpectations(t)
    bundle.Hooks.AssertExpectations(t)
    bundle.Events.AssertExpectations(t)
    bundle.Audit.AssertExpectations(t)
}

func TestRepairsService_Create_MissingVehicleID(t *testing.T) {
    bundle := testutil.NewMockCore(t)
    svc := repairs.NewService(bundle.Core, nil) // repo will not be called

    _, err := svc.Create(context.Background(), repairs.CreateRepairInput{
        Description: "Brake replacement",
        // VehicleID intentionally missing
    })

    assert.Error(t, err)
    var coreErr *core.Error
    assert.True(t, errors.As(err, &coreErr))
    assert.Equal(t, 422, coreErr.HTTPStatus())

    // Permission check should NOT have been called
    bundle.Permission.AssertNotCalled(t, "MustCan")
}

func TestRepairsService_Create_PermissionDenied(t *testing.T) {
    bundle := testutil.NewMockCore(t)

    bundle.Permission.On("MustCan", mock.Anything, "repairs.create").
        Panic(core.NewPermissionError("repairs.create"))

    svc := repairs.NewService(bundle.Core, nil)

    assert.Panics(t, func() {
        _, _ = svc.Create(context.Background(), repairs.CreateRepairInput{
            VehicleID:   uuid.New(),
            Description: "Brake replacement",
        })
    })
}
```

## Integration Tests

Integration tests run against a real PostgreSQL database in a dedicated test schema. They test the repository layer in full — actual SQL generation, type scanning, and constraint enforcement.

### Test Setup and Teardown

```go
// internal/modules/domain/repairs/repository_integration_test.go
package repairs_test

import (
    "context"
    "testing"

    "github.com/stretchr/testify/suite"
    "github.com/trazo-dev/trazo-core-service/internal/core/testutil"
    "github.com/trazo-dev/trazo-core-service/internal/modules/domain/repairs"
)

type RepositoryIntegrationSuite struct {
    suite.Suite
    db   *testutil.TestDB
    core *core.Core
    repo *repairs.Repository
}

func (s *RepositoryIntegrationSuite) SetupSuite() {
    // Create an isolated test schema and run migrations
    s.db = testutil.NewTestDB(s.T())
    s.core = testutil.NewTestCore(s.T(), s.db)
    s.repo = repairs.NewRepository(s.core)
}

func (s *RepositoryIntegrationSuite) TearDownSuite() {
    // Drop the test schema
    s.db.Cleanup()
}

func (s *RepositoryIntegrationSuite) SetupTest() {
    // Truncate tables before each test for isolation
    s.db.TruncateAll(s.T())
}

func TestRepositoryIntegration(t *testing.T) {
    if testing.Short() {
        t.Skip("skipping integration tests with -short")
    }
    suite.Run(t, new(RepositoryIntegrationSuite))
}
```

### Repository Integration Test Example

```go
func (s *RepositoryIntegrationSuite) TestCreateAndGetByID() {
    ctx := testutil.ContextWithUser(s.core, testutil.FixtureUser())

    input := repairs.CreateRepairInput{
        VehicleID:   uuid.New(),
        Description: "Engine overhaul",
    }

    created, err := s.db.WithTx(ctx, func(tx core.Tx) (*repairs.Repair, error) {
        return s.repo.CreateTx(ctx, tx, input)
    })
    s.Require().NoError(err)
    s.Require().NotNil(created)
    s.Equal("pending", created.Status)
    s.Equal("Engine overhaul", created.Description)

    fetched, err := s.repo.GetByID(ctx, created.ID.String())
    s.Require().NoError(err)
    s.Equal(created.ID, fetched.ID)
}

func (s *RepositoryIntegrationSuite) TestSoftDelete_HidesFromList() {
    ctx := testutil.ContextWithUser(s.core, testutil.FixtureUser())

    repair := s.createFixtureRepair(ctx)

    _ = s.db.WithTxVoid(ctx, func(tx core.Tx) error {
        return s.repo.SoftDeleteTx(ctx, tx, repair.ID.String())
    })

    list, total, err := s.repo.List(ctx, repairs.ListFilter{Limit: 20})
    s.Require().NoError(err)
    s.Equal(0, total)
    s.Empty(list)
}
```

<Aside type="note">
  Run integration tests with `go test ./... -run Integration` or use the build tag `//go:build integration` to separate them from unit tests in CI. Skip them with `go test -short ./...`.
</Aside>

## End-to-End Tests

E2E tests spin up a real HTTP server against a test database and fire actual HTTP requests. They verify the complete request lifecycle including middleware, Huma binding, service logic, and database persistence.

```go
// internal/modules/domain/repairs/handler_e2e_test.go
package repairs_test

import (
    "bytes"
    "encoding/json"
    "net/http"
    "net/http/httptest"
    "testing"

    "github.com/stretchr/testify/assert"
    "github.com/trazo-dev/trazo-core-service/internal/core/testutil"
    "github.com/trazo-dev/trazo-core-service/internal/modules/domain/repairs"
)

func TestHandler_CreateRepair_E2E(t *testing.T) {
    if testing.Short() {
        t.Skip("skipping e2e tests with -short")
    }

    // Build a real server with all middleware
    srv := testutil.NewTestServer(t, repairs.New())
    defer srv.Close()

    // Authenticate as a user with repairs.create permission
    token := testutil.IssueToken(t, srv.Core, testutil.WithPermissions("repairs.create"))

    body := map[string]any{
        "vehicle_id":  uuid.New().String(),
        "description": "Oil change",
    }
    bodyBytes, _ := json.Marshal(body)

    req, _ := http.NewRequest(http.MethodPost, srv.URL+"/api/v1/repairs", bytes.NewReader(bodyBytes))
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("Authorization", "Bearer "+token)

    resp, err := http.DefaultClient.Do(req)
    assert.NoError(t, err)
    assert.Equal(t, http.StatusOK, resp.StatusCode)

    var result repairs.Repair
    json.NewDecoder(resp.Body).Decode(&result)
    assert.Equal(t, "pending", result.Status)
    assert.Equal(t, "Oil change", result.Description)
}

func TestHandler_CreateRepair_E2E_Unauthorized(t *testing.T) {
    if testing.Short() {
        t.Skip("skipping e2e tests with -short")
    }

    srv := testutil.NewTestServer(t, repairs.New())
    defer srv.Close()

    // Token with no permissions
    token := testutil.IssueToken(t, srv.Core, testutil.WithPermissions())

    body := map[string]any{"vehicle_id": uuid.New().String(), "description": "Oil change"}
    bodyBytes, _ := json.Marshal(body)

    req, _ := http.NewRequest(http.MethodPost, srv.URL+"/api/v1/repairs", bytes.NewReader(bodyBytes))
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("Authorization", "Bearer "+token)

    resp, _ := http.DefaultClient.Do(req)
    assert.Equal(t, http.StatusForbidden, resp.StatusCode)
}
```

## Test File Naming Conventions

| File | Purpose |
|---|---|
| `service_test.go` | Unit tests for service methods using mocked Core |
| `handler_test.go` | Unit tests for handler request/response mapping |
| `repository_integration_test.go` | Integration tests for repository with real database |
| `handler_e2e_test.go` | End-to-end tests for the full HTTP cycle |
| `fixtures_test.go` | Shared test fixtures and helper factories |

---

## Test Helper Utilities

| Helper | Description |
|---|---|
| `testutil.NewMockCore(t)` | Constructs a `MockCoreBundle` with all mocks initialized |
| `testutil.NewTestDB(t)` | Creates an isolated PostgreSQL test schema with migrations applied |
| `testutil.NewTestServer(t, modules...)` | Starts an in-process HTTP server with the given modules wired |
| `testutil.IssueToken(t, core, opts...)` | Mints a valid JWT with configurable permissions and user identity |
| `testutil.ContextWithUser(core, user)` | Returns a context carrying the given user's session |
| `testutil.FixtureUser()` | Returns a deterministic user struct for use in tests |
| `testutil.NewNopLogger()` | Returns a logger implementation that discards all output |

## Running Tests

```bash
# Unit tests only (fast, no database required)
go test -short ./internal/modules/domain/repairs/...

# All tests including integration and e2e (requires a running database)
go test ./internal/modules/domain/repairs/...

# With race detector
go test -race ./internal/modules/domain/repairs/...

# Coverage report
go test -coverprofile=coverage.out ./internal/modules/domain/repairs/...
go tool cover -html=coverage.out
```
